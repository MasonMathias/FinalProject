# Firestore Security Rules for Development
# 
# IMPORTANT: These rules allow anyone to read/write data
# ONLY use these for development/testing!
# 
# To use these rules:
# 1. Go to Firebase Console
# 2. Firestore Database > Rules tab
# 3. Copy and paste the rules below
# 4. Click "Publish"

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // TEMPORARY DEVELOPMENT RULES - Allows all read/write
    // Replace with secure rules when authentication is implemented
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
